# Test Hub: Happy and Sad
import display, time

# see picture documentation

for i in range(4):
	display.display_show_pictogram(3) # pic 3 is HAPPY
	time.sleep(0.5)
	display.display_show_pictogram(5) # pic 5 is SAD
	time.sleep(0.5)
display.display_clear()


# Move Motors: Medium Motors on Ports A and B

motor.motor_move_at_speed(port.PORTB, 5000)
time.sleep(5)
motor.motor_stop()
